        s = 10

         function BinarySearch(arr,s){

            //Midpoint 
            let start = 0
            let end =   n
            while(start<=end){
                let middle = Math.floor((start+end)/2)   //middle (0 + 9)/2 = 4

                if(arr[middle] === s){
                    return 'At Index no ' + middle + " Data found"
                }else if(s > arr[middle]){
                    start = middle + 1
                }else{
                    end = middle-1
                }

            }

            return "Data not found..."

        }

        result = BinarySearch(arr,s)

        console.log(result)
