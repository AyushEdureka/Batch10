<body>
    <h1>Click a Button To Change the theme</h1>
    <button id="redButton">Red-Theme</button>
    <button id="blueButton">Blue-Theme</button>
    <button id="greenButton">Green-Theme</button>

    <marquee style="background-color: white; padding: 10px;">NOTICE : FLASH WARNING</marquee>

    <button id="reset">Reset</button>


    <script>
     //Selecting Elementing
       const redButton   =  document.getElementById('redButton')
       const blueButton =   document.getElementById('blueButton')
       const greenButton =  document.getElementById('greenButton')
       const reset   =       document.getElementById('reset')


       //Syntax for Events-1. TagName - Where you will perform action
       //       2. Add EventListner- Will give you the msg when user does the action
       //       3. What Type of Event('click')
       //       4. Function - To Do Task

       redButton.addEventListener('click',function(){
           
         document.body.style.backgroundColor="red"

       })


       blueButton.addEventListener('click',function(){
        document.body.style.backgroundColor="blue"
       })

       greenButton.addEventListener('click',function(){
        document.body.style.backgroundColor="green"
       })

       
       reset.addEventListener('click',function(){
        document.body.style.backgroundColor="white"
       })

    </script>    
</body>
