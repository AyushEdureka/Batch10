    <script>
        //Tree : Requirement - Node --> Define Root

        class TreeNode{  //Blueprint or structure of Single Node

            constructor(value){
               this.value = value
               this.children = []
            }

            addChild(ChildNode){
              //- Push Add Children in array
              this.children.push(ChildNode)
            }

        }

        //Family Tree

        //Object- new ClassName

        let grandparent = new TreeNode("GP - RootNode")

        let parent1 = new TreeNode("Parent -1 ")

        let parent2 = new TreeNode("Parent 2")

        let child1 = new TreeNode("Child 1")

        let child2 = new TreeNode("Child 2")

        //Connection
        grandparent.addChild(parent1)
        grandparent.addChild(parent2)

        parent1.addChild(child1)

        parent2.addChild(child2)

        console.log(grandparent)

    </script>
