    <div class="container">
        <h1>My To-Do-List</h1>

       <div class="input-section">
        <input type="text" id="taskInput" placeholder="Enter a new task...">
        <button id="addBtn">Add Task(push)</button>
       </div>

      <button id="removeLast">Remove Last (pop)</button>       
      <button id="addStart">Add at start(unshift)</button>
      <button id="removeFirst">Remove First(shift)</button>
      <button id="replaceTask">Replace Task(splice)</button>
      <button id="uppercaseTasks">Uppercase All(map)</button>
      <button id="findTask">Find Task</button>
      <button id="filterTasks">Filter Tasks</button>

      <ul id="taskList"></ul>

    </div>
    <script>
        let tasks = ["Buy grocies","Finish Project","Clean room"]

        //Display all Tasks
        function displayTasks(tasksArray=tasks){
          
            const list = document.getElementById("taskList");
            list.innerHTML = ""

            //forEach 
            tasksArray.forEach(function(task,index){
                const li = document.createElement("li")
                li.innerHTML = `<span>${index + 1}. ${task} </span>
                <button class="delete-btn" onclick="deleteTask(${index})">‚ùå</button>
                `;
                list.appendChild(li)

               
            });

        }
        

        //delee task using splice()
        function deleteTask(index){
            tasks.splice(index,1);
            displayTasks();
        }

       //Add new Task (push)
       document.getElementById("addBtn").onclick = function(){
        let newTask = document.getElementById("taskInput").value.trim()

        if(newTask){
            tasks.push(newTask)
            document.getElementById("taskInput").value = ""
            displayTasks();
        }else{
            alert("Please enter a task!")
        }
       }

       //Remove last data
       document.getElementById("removeLast").onclick = function(){
        tasks.pop()
        displayTasks()
       }

       //Add task at start
       document.getElementById("addStart").onclick = function(){
        let newTask = prompt("Enter a task to add at start")
        if(newTask){
            tasks.unshift(newTask)
            displayTasks()
        }
       }


       //Remove first data
        document.getElementById("removeFirst").onclick = function(){
        tasks.shift()
        displayTasks()
       }


       //Taskfrom all tasks (map)
       document.getElementById("uppercaseTasks").onclick = function(){
        tasks = tasks.map(task => task.toUpperCase())
        displayTasks()
       }


       //Find a task
       document.getElementById("findTask").onclick = function(){
        let keyword = prompt("Enter word to find in tasks :")
        let found = tasks.find(task => task.toLowerCase().includes(keyword.toLowerCase()))

        if(found){
            alert("Found Task : " + found)
        }else{
            alert("No Matching task found!")
        }
       }

       //Filter task(tasks)

       document.getElementById("filterTasks").onclick = function(){
        let filtered = tasks.filter(task => task.length > 5)
        displayTasks(filtered)
       };


       //Initial
       displayTasks()
    </script>
