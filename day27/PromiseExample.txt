        //Other Task : Async Process


        function OrderPizza(name){
     
            return new Promise((resolve,reject)=>{

             let pizzaReady = false;

             console.log(`Pizza for ${name}`)

             setTimeout(()=>{
                if(pizzaReady){
                    resolve("Pizza Delivered")
                }else{
                    reject("Order Cancalled")
                }
             })

            },5000)

        }

        OrderPizza("Lokesh")

        .then(message=>{
            console.log(message)
        })

        .catch(err=>{
            console.log(err)
        })

<img id="dogImage" src="" height="200" width="200">


    <script>
    
    function fetchDog(){
        //Step 1: fetch the Data from Api

        fetch('https://dog.ceo/api/breeds/image/random')

        //Step 2 : convert into json format data
        .then(response=>response.json())

        .then(data=>{
            document.getElementById("dogImage").src = data.message
        })

        .catch(error=>console.log("Error",error))
    }




    </script>
