       function loadMovie(movieName){
            
        return new Promise((resolve,reject)=>{
              console.log(`Loading ${movieName}...`)


              setTimeout(()=>{
                      resolve(`${movieName} is ready to watch..`)
              },2000)

        })

       }


       //Async - Which help us to create asynchornous function
       //await - Handle Waiting Time/Period

       async function watchMovie(){
              
        console.log("Getting ready to watch The Movie")

        //wait for movie to load
        let movieMessage = await loadMovie("Avengers")

        console.log(movieMessage)

        console.log("Enjoy the movie")

       }
